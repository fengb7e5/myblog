配置网络关闭 selinux 和防火墙
# 

控制节点
vim /etc/sysconfig/network-scripts/ifcfg-eth0

BOOTPROTO=static
ONBOOT=yes
IPADDR=192.168.10.10
NETMASK=255.255.255.0
GATEWAY=192.168.10.1
DNS=8.8.8.8

BOOTPROTO=static
ONBOOT=yes
IPADDR=10.0.0.10
NETMASK=255.255.255.0

# 重载配置文件
systemctl restart network

# 关闭 selinux
setenforce 0

SELINUX=disabled

解决 BUG
# 

apched 版本大于或等于 2.4 需要添加以下信息
vim /etc/httpd/conf.d/00-placement-api.conf

<Directory /usr/bin>
   <IfVersion >= 2.4>
      Require all granted
   </IfVersion>
   <IfVersion < 2.4>
      Order allow,deny
      Allow from all
   </IfVersion>
</Directory>

# 重启 apched
systemctl restart httpd


# 验证
placement-status upgrade check
